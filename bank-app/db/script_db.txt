CREATE TABLE public.clients (
	client_id serial4 NOT NULL,
	"name" varchar(255) NOT NULL,
	kind_property varchar(255) NOT NULL,
	address varchar(255) NOT NULL,
	phone varchar(20) NOT NULL,
	contact varchar(255) NOT NULL,
	CONSTRAINT clients_pkey PRIMARY KEY (client_id)
);


-- public.kind_credit definition

-- Drop table

-- DROP TABLE public.kind_credit;

CREATE TABLE public.kind_credit (
	kind_credit_id serial4 NOT NULL,
	"name" varchar(255) NULL,
	conditions text NOT NULL,
	rate numeric NOT NULL,
	term int4 NOT NULL,
	CONSTRAINT kind_credit_pkey PRIMARY KEY (kind_credit_id)
);


-- public.credits definition

-- Drop table

-- DROP TABLE public.credits;

CREATE TABLE public.credits (
	credit_id serial4 NOT NULL,
	kind_credit_id int4 NOT NULL,
	client_id int4 NOT NULL,
	summa numeric NOT NULL,
	"date" date NOT NULL,
	CONSTRAINT credits_pkey PRIMARY KEY (credit_id),
	CONSTRAINT credits_client_id_fkey FOREIGN KEY (client_id) REFERENCES public.clients(client_id) ON DELETE CASCADE,
	CONSTRAINT credits_kind_credit_id_fkey FOREIGN KEY (kind_credit_id) REFERENCES public.kind_credit(kind_credit_id) ON DELETE CASCADE
);
